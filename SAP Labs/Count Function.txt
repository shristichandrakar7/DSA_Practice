#include<bits/stdc++.h>
using namespace std;

vector<int> d_distance(int N, string A, int D) {
    vector<int> result(N, 0);  // Initialize the result vector with zeros

    for (int i = 0; i < N; i++) {
        int char_i = A[i];

        for (int j = i + 1; j < N; j++) {
            int char_j = A[j];
            int abs_diff = abs(char_i - char_j);

            if (abs_diff <= D) {
                result[i]++;
            }
        }
    }

    return result;
}

int main() {
#ifndef ONLINE_JUDGE
    //for getting input from input.txt
    freopen("input.txt", "r", stdin);
    //for writing output to output.txt
    freopen("output.txt", "w", stdout);
#endif
    int T, N, D;
    cin >> T;

    while (T--) {
        cin >> N;
        string A;
        cin >> A;
        cin >> D;

        vector<int> counts = d_distance(N, A, D);

        for (int i = 0; i < N; i++) {
            cout << counts[i] << " ";
        }

        cout << endl;
    }

    return 0;
}
