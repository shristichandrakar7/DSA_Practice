Problem Statement: Given an array that contains only 1 and 0 return the count of maximum consecutive ones in the array.

Example 1:
Input: prices = {1, 1, 0, 1, 1, 1}
Output: 3
------------------------------------------------------------------------------------------------------------------------------------------
Approach 1:

TC : O(n)
SC : O(1)

int findMaxConsecutiveOnes(vector<int>& nums) {
	int n = nums.size();
	int maxi = 0;
	int cnt = 0;
	for(int i = 0 ; i < n; i++) {
		if(nums[i] == 1) cnt++;
		else cnt = 0;
		maxi = max(maxi, cnt);
	}
	return maxi;
}
